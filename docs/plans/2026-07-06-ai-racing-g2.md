# AI Racing (G2) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Turn the fixed-speed AI curve-followers into physics-driven opponents that actually race — with lap/progress tracking, live race positions in the HUD, difficulty selection + rubber-banding, and a results table comparing player vs AI finish times.

**Architecture:** Extract the player's force math into a shared, input-driven `applyKartForces` core so player and AI share one handling model. AI cars become dynamic RigidBodies driven by a pure-pursuit + curvature-braking controller that emits `{steer, throttle, brake, handbrake}`. A mutable `raceStandings` module (mirroring the `carTransform`/`keys` pattern) holds continuous progress for every car and is read by a logic-only `RaceDirector` that pushes only low-frequency values (`playerPosition`, `racerCount`, final `raceResults`) into the store at ~10 Hz. This keeps per-frame race math out of React, consistent with the Phase 1 perf work.

**Tech Stack:** React 19 + TypeScript, @react-three/fiber, @react-three/rapier (Rapier), Zustand, Tailwind/shadcn overlay.

**Testing note:** No test runner is configured (Vitest is deferred to Phase 4 · C6). Each task is gated on **behavioral verification** — driving the app with `npm run dev` and observing the specific behavior — plus `npm run build` (tsc) and `npm run lint` staying green. Pure functions (`sortStandings`, `progressFraction`, `curvatureTargetSpeed`) are written in isolation with exported signatures so C6 can unit-test them later.

**Key invariant:** Task 1 is a *pure refactor*. The player car must feel byte-for-byte identical before and after. Verify by driving (accelerate, top speed, cornering, handbrake drift, boost, turbo, respawn) before starting AI work.

```ts
import * as THREE from "three";
import type { RapierRigidBody } from "@react-three/rapier";
import { smoothAlpha } from "./smoothing";
import {
  ACCELERATION,
  BRAKE_FORCE,
  DECELERATION,
  MAX_REVERSE_SPEED,
  MAX_STEERING_ANGLE,
  STEERING_SPEED,
  CAR_MASS,
} from "./carConstants"; // NOTE: move these consts into carConstants.ts in this task (see Step 2)

export interface KartInput {
  steer: number;      // -1..1 (target, before lerp)
  throttle: number;   // 0..1
  brake: number;      // 0..1
  handbrake: boolean;
}

export interface KartTuning {
  maxSpeed: number;   // forward speed cap (m/s)
  accelMul: number;   // acceleration multiplier (boost/turbo folded in by caller if desired)
  gripBase: number;   // lateral grip strength when not handbraking (player: 0.85, grip-boost: 0.95)
}

export interface KartStepResult {
  forwardSpeed: number;
  speed: number;
  yaw: number;
  pos: { x: number; y: number; z: number };
}

const yawFromQuat = (q: { x: number; y: number; z: number; w: number }) =>
  Math.atan2(2 * (q.w * q.y + q.x * q.z), 1 - 2 * (q.y * q.y + q.z * q.z));

/**
 * Applies one frame of kart handling forces to `body` from an abstract input.
 * Owns: velocity decomposition, steering lerp + angvel, accel/brake/coast,
 * lateral grip correction, handbrake damping, upright slerp.
 * `steerState` is a per-car mutable ref-like holder: { current: number }.
 * Returns kinematics the caller needs (progress, camera, HUD).
 */
export function applyKartForces(
  body: RapierRigidBody,
  input: KartInput,
  steerState: { current: number },
  tuning: KartTuning,
  dt: number,
): KartStepResult {
  const vel = body.linvel();
  const rot = body.rotation();
  const yaw = yawFromQuat(rot);

  const forward = new THREE.Vector3(Math.sin(yaw), 0, Math.cos(yaw));
  const right = new THREE.Vector3(Math.cos(yaw), 0, -Math.sin(yaw));
  const forwardSpeed = vel.x * forward.x + vel.z * forward.z;
  const lateralSpeed = vel.x * right.x + vel.z * right.z;
  const speed = Math.sqrt(vel.x ** 2 + vel.z ** 2);

  // Steering lerp (frame-rate independent).
  const targetSteering = input.steer * MAX_STEERING_ANGLE;
  const steerLerp = 1 - Math.pow(0.001, dt);
  steerState.current += (targetSteering - steerState.current) * steerLerp;
  const currentSteering = steerState.current;

  // Longitudinal.
  let acceleration = 0;
  if (input.throttle > 0) {
    if (forwardSpeed < tuning.maxSpeed) {
      acceleration = ACCELERATION * tuning.accelMul * input.throttle;
    }
  } else if (input.brake > 0) {
    if (forwardSpeed > 0.5) acceleration = -BRAKE_FORCE * input.brake;
    else if (forwardSpeed > -MAX_REVERSE_SPEED)
      acceleration = -ACCELERATION * 0.5 * input.brake;
  } else {
    const damp = Math.exp(-DECELERATION * 0.15 * dt);
    body.setLinvel({ x: vel.x * damp, y: vel.y, z: vel.z * damp }, true);
  }
  if (Math.abs(acceleration) > 0.1) {
    const f = acceleration * dt * CAR_MASS;
    body.applyImpulse({ x: forward.x * f, y: 0, z: forward.z * f }, true);
  }

  // Steering angvel.
  if (
    Math.abs(currentSteering) > 0.01 &&
    (Math.abs(speed) > 0.1 || input.throttle > 0 || input.brake > 0)
  ) {
    const steerSign = forwardSpeed >= 0 ? 1 : -1;
    const speedFactor = Math.max(
      Math.min(speed / 15, 1) * Math.max(1 - speed / 120, 0.3),
      0.15,
    );
    body.setAngvel(
      { x: 0, y: -currentSteering * STEERING_SPEED * speedFactor * steerSign, z: 0 },
      true,
    );
  } else {
    body.setAngvel({ x: 0, y: 0, z: 0 }, true);
  }

  // Lateral grip.
  if (Math.abs(lateralSpeed) > 0.2) {
    const grip = input.handbrake ? 0.4 : tuning.gripBase;
    const c = -lateralSpeed * grip * CAR_MASS * dt;
    body.applyImpulse({ x: right.x * c, y: 0, z: right.z * c }, true);
  }
  if (input.handbrake) {
    body.setLinvel(
      { x: vel.x * (1 - 1.5 * dt), y: vel.y, z: vel.z * (1 - 1.5 * dt) },
      true,
    );
  }

  // Upright slerp.
  const upright = new THREE.Quaternion().setFromAxisAngle(
    new THREE.Vector3(0, 1, 0),
    yaw,
  );
  const cur = new THREE.Quaternion(rot.x, rot.y, rot.z, rot.w);
  cur.slerp(upright, smoothAlpha(0.3, dt));
  body.setRotation({ x: cur.x, y: cur.y, z: cur.z, w: cur.w }, true);

  const pos = body.translation();
  return { forwardSpeed, speed, yaw, pos };
}
```

```ts
// Fraction 0..1 around the centerline of the point nearest (x,z).
export function progressFraction(points: THREE.Vector3[], x: number, z: number): number {
  let best = 0, bestD = Infinity;
  for (let i = 0; i < points.length; i++) {
    const dx = points[i].x - x, dz = points[i].z - z;
    const d = dx * dx + dz * dz;
    if (d < bestD) { bestD = d; best = i; }
  }
  return best / points.length;
}
```

**Step 2: `raceStandings.ts`.**

```ts
export interface RacerState {
  id: string;
  label: string;      // "You", "#2", ...
  isPlayer: boolean;
  color: string;
  lap: number;        // 1-based current lap
  progress: number;   // lap - 1 + fractionAroundTrack (monotonic, continuous)
  finishTime: number | null;
  position: number;   // 1-based, filled by recompute()
}

export const raceStandings: { cars: RacerState[] } = { cars: [] };

/** Register the roster at race start (player + AI). Order/refs are stable. */
export function initStandings(cars: Omit<RacerState, "position">[]) {
  raceStandings.cars = cars.map((c) => ({ ...c, position: 0 }));
}

export function getRacer(id: string): RacerState | undefined {
  return raceStandings.cars.find((c) => c.id === id);
}

/** Continuous progress with wrap guard: 0.98 -> 0.02 must read as forward. */
export function updateProgress(id: string, lap: number, fraction: number) {
  const car = getRacer(id);
  if (!car) return;
  car.lap = lap;
  car.progress = lap - 1 + fraction;
}

/** Pure: sort a copy by finished-first (asc time) then progress desc; assign positions. */
export function sortStandings(cars: RacerState[]): RacerState[] {
  const ranked = [...cars].sort((a, b) => {
    if (a.finishTime !== null && b.finishTime !== null) return a.finishTime - b.finishTime;
    if (a.finishTime !== null) return -1;
    if (b.finishTime !== null) return 1;
    return b.progress - a.progress;
  });
  ranked.forEach((c, i) => (c.position = i + 1));
  return ranked;
}

export function recomputeStandings() {
  sortStandings(raceStandings.cars);
}
```

const LOOKAHEAD_PER_MS = 0.6; // extra meters per m/s of speed

**Step 2: `curvatureTargetSpeed` (pure).** Given the angle between the tangent now and a bit ahead, return a target forward speed: straight → `baseMax`, sharp bend → down to ~35% of it.

```ts
export function curvatureTargetSpeed(baseMax: number, turnAngleRad: number): number {
  const t = Math.min(Math.abs(turnAngleRad) / (Math.PI / 2), 1); // 0..1
  return baseMax * (1 - 0.65 * t);
}
```

**Step 3: `computeAiInput`.** Takes the AI body pose, the track curve + progress `t`, `baseMax` (already difficulty+rubber scaled by caller), and returns `KartInput`. Steer = clamped signed yaw error to the lookahead point; throttle/brake from `forwardSpeed` vs `curvatureTargetSpeed`.

```ts
import * as THREE from "three";
import type { KartInput } from "./kartForces";

export function computeAiInput(args: {
  curve: THREE.Curve<THREE.Vector3>;
  t: number;            // current progress fraction 0..1
  posX: number; posZ: number; yaw: number; forwardSpeed: number;
  baseMax: number;      // difficulty + rubberband already applied
  trackLength: number;  // curve.getLength(), for lookahead->t conversion
}): { input: KartInput; targetSpeed: number } {
  const { curve, t, posX, posZ, yaw, forwardSpeed, baseMax, trackLength } = args;
  const lookahead = LOOKAHEAD_BASE + LOOKAHEAD_PER_MS * Math.max(forwardSpeed, 0);
  const tAhead = (t + lookahead / trackLength) % 1;
  const aim = curve.getPoint(tAhead);

  // Signed yaw error to the aim point.
  const desiredYaw = Math.atan2(aim.x - posX, aim.z - posZ);
  let err = desiredYaw - yaw;
  err = Math.atan2(Math.sin(err), Math.cos(err)); // wrap to -pi..pi
  const steer = Math.max(-1, Math.min(1, err / 0.6));

  // Curvature: tangent now vs a bit further ahead.
  const tFar = (t + (lookahead * 1.6) / trackLength) % 1;
  const tanNow = curve.getTangent(t).normalize();
  const tanFar = curve.getTangent(tFar).normalize();
  const turn = Math.acos(Math.max(-1, Math.min(1, tanNow.dot(tanFar))));
  const targetSpeed = curvatureTargetSpeed(baseMax, turn);

  const throttle = forwardSpeed < targetSpeed ? 1 : 0;
  const brake = forwardSpeed > targetSpeed * 1.15 ? 1 : 0;
  return { input: { steer, throttle, brake, handbrake: false }, targetSpeed };
}

/** Rubber-band: nudge maxSpeed toward closing the gap to the player. */
export function rubberBand(baseMax: number, gap: number, rubberMax: number): number {
  // gap = playerProgress - aiProgress (positive => AI behind => speed up).
  const k = Math.max(-1, Math.min(1, gap * 2));
  return baseMax * (1 + rubberMax * k);
}
```
